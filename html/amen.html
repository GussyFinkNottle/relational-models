{- A datatype of notations for natural numbers, with accompanying reduction relation. 
-}
module _ where
  pow : Set -> Set₁
  pow A = A -> Set
  rel : (_ _ : Set) -> Set₁
  rel A B = A -> pow B

  module amen where

    data fm : Set where
         _:+:_ _:*:_ _:^:_ _:Ø:_ : (_ _ : fm)-> fm
         [+] [*] [^] [Ø] : fm
         [1] : fm

    data em : rel fm fm  where
        b0' : (a : fm)                                          -> em (a :^: [Ø]) [1] 
        b1' : (a x : fm)-> em a x                               -> em (a :^: [1]) x 
        be' : (a x : fm)-> em a x                               -> em (a :^: [^]) (x :^: [^]) 
        bm' : (a x : fm)-> em a x                               -> em (a :^: [*]) (x :^: [*]) 
        ba' : (a x : fm)-> em a x                               -> em (a :^: [+]) (x :^: [+]) 
        be'' : (a x : fm)-> em a x                              -> em (a :*: [^]) (x :*: [^]) 
        bm'' : (a x : fm)-> em a x                              -> em (a :*: [*]) (x :*: [*]) 
        ba'' : (a x : fm)-> em a x                              -> em (a :*: [+]) (x :*: [+])
        ww   : (a x : fm)-> em a x                              -> em (a :*: [1]) x
        ww'  : (a x : fm)                                       -> em (a :*: [Ø]) [Ø]
        be''' : (a x : fm)-> em a x                             -> em (a :+: [^]) (x :+: [^]) 
        bm''' : (a x : fm)-> em a x                             -> em (a :+: [*]) (x :+: [*]) 
        ba''' : (a x : fm)-> em a x                             -> em (a :+: [+]) (x :+: [+])
        ba'''' : (a x : fm)-> em a x                            -> em (a :+: [Ø]) x
        ba''''' : (a x : fm)-> em a x                           -> em (a :+: [1]) (x :+: [1])
        te : (a b x y : fm)-> em b x -> em (a :^: x) y          -> em (a :^: b) y
        tm : (a b x y : fm)-> em b x -> em (a :*: x) y          -> em (a :*: b) y
        ta : (a b x y : fm)-> em b x -> em (a :+: x) y          -> em (a :+: b) y
        t0 : (a b x y : fm)-> em b y                            -> em (a :Ø: b) y
        ra : (a b c y : fm)-> em ((c :^: a) :*: (c :^: b)) y    -> em (c :^: (a :+: b)) y 
        rm : (a b c y : fm)-> em ((c :^: a) :^: b) y            -> em (c :^: (a :*: b)) y
        da : (a b c y : fm)-> em ((c :*: a) :+: (c :*: b)) y    -> em (c :*: (a :+: b)) y   -- dist * / +
        d0 : (c : fm)                                           -> em (c :*: [Ø]) [Ø]
        mm : (a b c y : fm)-> em ((a :*: b) :*: c) y            -> em (a :*: (b :*: c)) y   -- * monoid
        m1  : (a y : fm)-> em a y                               -> em (a :*: [1]) y
        m1' : (b y : fm)-> em b y                               -> em ([1] :*: b) y
        ma  : (a b c y : fm)-> em ((a :+: b) :+: c) y           -> em (a :+: (b :+: c)) y   -- + monoid
        m0  : (a y : fm)-> em a y                               -> em (a :+: [Ø]) y
        m0' : (a y : fm)-> em a y                               -> em ([Ø] :+: a) y
        m0'' : (a y : fm)-> em a y                              -> em (a :+: [1]) (y :+: [1]) 

        ie : (a b  y : fm)-> em (b :^: a) y                    -> em (a :^: (b :^: [^]))  y
        im : (a b  y : fm)-> em (b :*: a) y                    -> em (a :^: (b :^: [*]))  y
        ia : (a b  y : fm)-> em (b :+: a) y                    -> em (a :^: (b :^: [+]))  y
        i0 : (a b  y : fm)-> em a y                            -> em (a :^: (b :^: [Ø]))  y
        be : em [^] [^]
        bm : em [*] [*]
        ba : em [+] [+]
        bØ : em [Ø] [Ø]
        b1 : em [1] [1]
        zz : (a b c x y : fm)->  em a x -> em b [1] -> em (a :+: b) (x :+: [1])
--         q  : (x : fm)-> em x x ->

    diag : (x : fm)-> Set
    diag x = em x x 
    t t' : diag ([^] :^: [^]) 
    t = te [^] [^] [^] ([^] :^: [^]) be (be' [^] [^] be) 
    t' = be' [^] [^] be
    mmm : diag ([*] :*: [*])
    mmm = bm'' [*] [*] bm
    aaa : diag ([+] :+: [+])
    aaa = ba''' [+] [+] ba 

    one two three four four' four'' : fm
    one = [1] -- [Ø] :^: [Ø]
    two = one :+: one
    three = two :+: one
    four  = two :*: two
    four' = two :^: two
    four'' = two :+: two

    t''' : em one [1]
    t''' = b1 -- b0' [Ø]
    t'''' : em two ([1] :+: [1])
    t'''' = ta one one [1] ([1] :+: [1]) t''' (zz one [1] [1] [1] [1] t''' b1)
    t''''' : em (two :+: one) (([1] :+: [1]) :+: [1])
    t''''' = zz two one [1] ([1] :+: [1]) [1] t'''' t''' 

    t'' : em four'' ((([1] :+: [1]) :+: [1]) :+: [1])
    t'' = ma two one one ((([1] :+: [1]) :+: [1]) :+: [1])
                         (ta (two :+: one) one [1] ((([1] :+: [1]) :+: [1]) :+: [1]) t'''
                                                               (zz (two :+: one) [1] [1] (([1] :+: [1]) :+: [1]) [1] t''''' b1))
